    <script>
        let selectedFormat = null;
        let currentUrl = '';
        let allFormats = [];
        let currentFilter = 'all';

        function showError(message) {
            const alert = document.getElementById('errorAlert');
            alert.innerHTML = `<i class="fas fa-exclamation-circle"></i> ${message}`;
            alert.style.display = 'block';
            setTimeout(() => alert.style.display = 'none', 6000);
        }

        function showSuccess(message) {
            const alert = document.getElementById('successAlert');
            alert.innerHTML = `<i class="fas fa-check-circle"></i> ${message}`;
            alert.style.display = 'block';
            setTimeout(() => alert.style.display = 'none', 6000);
        }

        function clearForm() {
            document.getElementById('videoUrl').value = '';
            document.getElementById('videoInfo').style.display = 'none';
            selectedFormat = null;
            allFormats = [];
            currentFilter = 'all';
            document.querySelectorAll('.tab').forEach((tab, i) => {
                tab.classList.toggle('active', i === 0);
            });
        }

        async function getVideoInfo() {
            const url = document.getElementById('videoUrl').value.trim();
            
            if (!url) {
                showError('Please enter a valid YouTube URL');
                return;
            }

            if (!url.includes('youtube.com') && !url.includes('youtu.be')) {
                showError('Please enter a valid YouTube URL');
                return;
            }

            currentUrl = url;
            document.getElementById('loading').style.display = 'block';
            document.getElementById('videoInfo').style.display = 'none';

            try {
                const response = await fetch('/get_video_info', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ url: url })
                });

                const data = await response.json();

                if (!response.ok) {
                    throw new Error(data.error || 'Failed to fetch video information');
                }

                allFormats = data.formats;
                displayVideoInfo(data);
            } catch (error) {
                showError(error.message);
            } finally {
                document.getElementById('loading').style.display = 'none';
            }
        }

        function displayVideoInfo(data) {
            const preview = document.getElementById('videoPreview');
            const duration = formatDuration(data.duration);
            const views = formatNumber(data.view_count);

            preview.innerHTML = `
                <div class="thumbnail-wrapper">
                    <img src="${data.thumbnail}" alt="Video Thumbnail">
                    <div class="play-overlay">
                        <i class="fas fa-play"></i>
                    </div>
                </div>
                <div class="video-details">
                    <h3>${escapeHtml(data.title)}</h3>
                    <div class="video-stats">
                        <div class="stat-item">
                            <i class="fas fa-user-circle"></i>
                            <span>${escapeHtml(data.uploader)}</span>
                        </div>
                        <div class="stat-item">
                            <i class="fas fa-clock"></i>
                            <span>${duration}</span>
                        </div>
                        <div class="stat-item">
                            <i class="fas fa-eye"></i>
                            <span>${views} views</span>
                        </div>
                    </div>
                </div>
            `;

            filterFormats('all');
            document.getElementById('videoInfo').style.display = 'block';
        }

        function filterFormats(type, element) {
            currentFilter = type;
            
            // Update active tab
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            if (element) {
                element.classList.add('active');
            } else {
                document.querySelector('.tab').classList.add('active');
            }

            const grid = document.getElementById('formatsGrid');
            grid.innerHTML = '';

            let filteredFormats = allFormats;

            if (type === 'video') {
                filteredFormats = allFormats.filter(f => f.vcodec !== 'none' && f.acodec !== 'none');
            } else if (type === 'audio') {
                filteredFormats = allFormats.filter(f => f.acodec !== 'none' && f.vcodec === 'none');
            } else if (type === 'hd') {
                filteredFormats = allFormats.filter(f => {
                    const res = f.resolution.toLowerCase();
                    return res.includes('1080') || res.includes('720') || res.includes('2160') || res.includes('4k');
                });
            }

            // Remove duplicates
            const uniqueFormats = [];
            const seen = new Set();

            filteredFormats.forEach(format => {
                const key = `${format.resolution}-${format.ext}`;
                if (!seen.has(key)) {
                    seen.add(key);
                    uniqueFormats.push(format);
                }
            });

            if (uniqueFormats.length === 0) {
                grid.innerHTML = '<p style="text-align: center; color: #999; padding: 40px;">No formats found for this filter</p>';
                return;
            }

            uniqueFormats.forEach(format => {
                const card = document.createElement('div');
                card.className = 'format-card';
                card.onclick = () => selectFormat(format.format_id, card);
                
                const filesize = format.filesize ? formatFileSize(format.filesize) : 'Unknown';
                const isHD = format.resolution.includes('1080') || format.resolution.includes('720') || format.resolution.includes('2160');
                const isAudio = format.vcodec === 'none';
                
                card.innerHTML = `
                    ${isHD ? '<span class="quality-badge"><i class="fas fa-crown"></i> HD</span>' : ''}
                    <div class="format-title">
                        <i class="fas fa-${isAudio ? 'music' : 'video'}"></i>
                        ${format.resolution}
                    </div>
                    <div class="format-details">
                        <strong>Format:</strong> ${format.ext.toUpperCase()}<br>
                        <strong>Size:</strong> ${filesize}<br>
                        <strong>Type:</strong> ${isAudio ? 'Audio Only' : 'Video + Audio'}
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        function selectFormat(formatId, element) {
            document.querySelectorAll('.format-card').forEach(card => {
                card.classList.remove('selected');
            });
            element.classList.add('selected');
            selectedFormat = formatId;
            document.getElementById('downloadBtn').disabled = false;
        }

        async function downloadVideo() {
            if (!selectedFormat) {
                showError('Please select a format first');
                return;
            }

            const btn = document.getElementById('downloadBtn');
            const progressBar = document.getElementById('downloadProgress');
            const progressFill = document.getElementById('downloadProgressFill');
            
            btn.disabled = true;
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Preparing Download...';
            progressBar.style.display = 'block';
            progressFill.style.width = '30%';

            try {
                const response = await fetch('/download', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        url: currentUrl,
                        format_id: selectedFormat
                    })
                });

                const data = await response.json();

                if (!response.ok) {
                    throw new Error(data.error || 'Download failed');
                }

                progressFill.style.width = '100%';
                showSuccess('Download started! Your file will be downloaded shortly.');
                
                setTimeout(() => {
                    window.location.href = `/download_file/${data.filename}`;
                    progressBar.style.display = 'none';
                    progressFill.style.width = '0%';
                }, 1000);
                
            } catch (error) {
                showError(error.message);
                progressBar.style.display = 'none';
            } finally {
                btn.disabled = false;
                btn.innerHTML = '<i class="fas fa-download"></i> Download Selected Format';
            }
        }

        function formatDuration(seconds) {
            if (!seconds) return 'N/A';
            const hours = Math.floor(seconds / 3600);
            const minutes = Math.floor((seconds % 3600) / 60);
            const secs = seconds % 60;
            
            if (hours > 0) {
                return `${hours}:${String(minutes).padStart(2, '0')}:${String(secs).padStart(2, '0')}`;
            }
            return `${minutes}:${String(secs).padStart(2, '0')}`;
        }

        function formatNumber(num) {
            if (!num) return 'N/A';
            if (num >= 1000000) {
                return (num / 1000000).toFixed(1) + 'M';
            }
            if (num >= 1000) {
                return (num / 1000).toFixed(1) + 'K';
            }
            return num.toLocaleString();
        }

        function formatFileSize(bytes) {
            if (!bytes) return 'Unknown';
            const mb = bytes / (1024 * 1024);
            if (mb >= 1000) {
                return (mb / 1024).toFixed(2) + ' GB';
            }
            return mb.toFixed(2) + ' MB';
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
